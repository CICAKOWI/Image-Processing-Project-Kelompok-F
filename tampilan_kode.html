01 Grayscale — Konversi RGB ke Grayscale (rata-rata R+G+B) 
def grayscale(pic):
    row, col = pic.shape[0], pic.shape[1]
    pic_gs = np.zeros(shape=(row, col), dtype=np.uint8)

    r = pic[:, :, 0].astype(np.float32)
    g = pic[:, :, 1].astype(np.float32)
    b = pic[:, :, 2].astype(np.float32)

    average = (r + g + b) / 3
    pic_gs[:, :] = average[:, :]

    return pic_gs
02 Resizing (Down Pooling) — Resize ke 28×28 dengan Average/Max Pooling 
# Parameter ukuran target
m, n = 28, 28   # ukuran output: 28x28 piksel

def down_pooling(pic, m, n, mode_pooling):
    row_asal, col_asal = pic.shape
    row_hasil, col_hasil = m, n

    pic_res = np.zeros(shape=(row_hasil, col_hasil), dtype=np.float32)

    for i_res in range(0, row_hasil):
        for j_res in range(0, col_hasil):
            r_start = int(i_res * row_asal / row_hasil)
            r_end   = int((i_res + 1) * row_asal / row_hasil)
            c_start = int(j_res * col_asal / col_hasil)
            c_end   = int((j_res + 1) * col_asal / col_hasil)

            if r_end <= r_start: r_end = r_start + 1
            if c_end <= c_start: c_end = c_start + 1

            if mode_pooling == "average":
                counter, jumlah = 0, 0
                for k in range(r_start, r_end):
                    for l in range(c_start, c_end):
                        jumlah  += int(pic[k, l])
                        counter += 1
                pic_res[i_res, j_res] = jumlah / counter

            if mode_pooling == "max":
                nilai_max = 0
                for k in range(r_start, r_end):
                    for l in range(c_start, c_end):
                        if pic[k, l] > nilai_max:
                            nilai_max = pic[k, l]
                pic_res[i_res, j_res] = nilai_max

    # Kembalikan ke nilai binary 0 dan 255
    pic_final = np.zeros(shape=(row_hasil, col_hasil), dtype=np.uint8)
    for i in range(0, row_hasil):
        for j in range(0, col_hasil):
            pic_final[i, j] = 255 if pic_res[i, j] > 0 else 0

    return pic_final
03 Binarization — Otsu Threshold + Manual Threshold (th=90) 
# Otsu Thresholding (dihitung otomatis dari histogram)
def otsu_threshold(pic_gs):
    hist = np.zeros(shape=(256), dtype=np.float64)
    row, col = pic_gs.shape

    for i in range(0, row):
        for j in range(0, col):
            hist[pic_gs[i, j]] += 1

    total     = row * col
    sum_total = sum(i * hist[i] for i in range(256))
    sum_bg, w_bg, var_max, th = 0, 0, -1, 127

    for i in range(256):
        w_bg += hist[i]
        if w_bg == 0: continue
        w_fg = total - w_bg
        if w_fg == 0: break
        sum_bg   += i * hist[i]
        mean_bg   = sum_bg / w_bg
        mean_fg   = (sum_total - sum_bg) / w_fg
        var_between = w_bg * w_fg * ((mean_bg - mean_fg) ** 2)
        if var_between > var_max:
            var_max, th = var_between, i
    return th

# Binarisasi: pilih threshold terkecil (Otsu vs Manual)
th_manual = 90

def binarisasi(pic_gs):
    th_otsu = otsu_threshold(pic_gs)
    th = th_manual if th_otsu > th_manual else th_otsu

    row, col = pic_gs.shape
    pic_bin = np.zeros(shape=(row, col), dtype=np.uint8)

    for i in range(0, row):
        for j in range(0, col):
            # Objek = putih (255), background = hitam (0)
            pic_bin[i, j] = 255 if pic_gs[i, j] < th else 0

    return pic_bin, th
04 Create Dataset — Buat file .npy dari gambar hasil preprocessing 
import numpy as np
import matplotlib.pyplot as plt
import os

mode_pooling = input("Mode pooling (average/max): ").lower().strip()

dir_in  = os.path.join(r"d:\...\01_ImageProcessing", mode_pooling)
dir_out = os.path.join(r"d:\...\02a_Create_Dataset", mode_pooling)
os.makedirs(dir_out, exist_ok=True)

# Daftar nama kelas (12 kelas zodiak)
kelas = ["Aquarius (", "Aries (", "Cancer (", "Capricorn (",
         "Gemini (",   "Leo (",   "Libra (",  "Pisces (",
         "Sagitarius (", "Scorpio (", "Taurus (", "Virgo ("]
ext = ")_ready.jpg"

juml_kelas   = 12
juml_sampel  = juml_kelas * 10   # 120 sampel total

# Baca 1 gambar untuk dapatkan dimensi (28x28)
pic      = plt.imread(os.path.join(dir_in, kelas[0] + "0" + ext))
row, col = pic.shape[0], pic.shape[1]
juml_kolom = row * col + 1        # +1 untuk kolom nomor urut

# Buat template dataset
dataset = np.zeros(shape=(juml_sampel, juml_kolom), dtype=np.uint16)

# Isi kolom-0 dengan nomor urut
for k in range(juml_sampel):
    dataset[k, 0] = k

# Masukkan nilai piksel tiap gambar ke dataset
k = 0
for i, nama in enumerate(kelas):
    for n in range(10):
        pic = plt.imread(os.path.join(dir_in, nama + str(n) + ext))
        if pic.dtype != np.uint8:
            pic = (pic * 255).astype(np.uint8)
        temp = pic[:, :, 0] if len(pic.shape) == 3 else pic
        dataset[k, 1:] = temp.reshape(row * col)
        k += 1

# Simpan dataset ke file .npy
nama_file = f"inputs_{juml_sampel}_{juml_kolom}.npy"
np.save(os.path.join(dir_out, nama_file), dataset)
print(f"Dataset disimpan: {nama_file}  |  Shape: {dataset.shape}")
05 Randomize Dataset — Acak urutan data inputs & labels secara sinkron 
import numpy as np
import os

mode_pooling = input("Mode pooling (average/max): ").lower().strip()

dir_inputs = os.path.join(r"d:\...\02a_Create_Dataset", mode_pooling)
dir_labels = os.path.join(r"d:\...\02b_Create_Label",   mode_pooling)
dir_out    = os.path.join(r"d:\...\03_Randomize",       mode_pooling)
os.makedirs(dir_out, exist_ok=True)

# Load inputs dan labels
inputs = np.load(os.path.join(dir_inputs, "inputs_120_785.npy"))
labels = np.load(os.path.join(dir_labels, "labels_120_13.npy"))

m, n = np.shape(inputs)   # 120 baris, 785 kolom
r, s = np.shape(labels)   # 120 baris, 13 kolom

# Buat array indeks [0, 1, 2, ..., 119] lalu acak
a = list(range(m))
np.random.shuffle(a)

print("Indeks sebelum acak :", 0, "→", m-1)
print("Indeks setelah acak  :", a[0], "→", a[m-1])

# Susun ulang inputs dan labels dengan indeks yang sama
inputs_random = np.zeros(shape=(m, n), dtype=float)
labels_random = np.zeros(shape=(r, s), dtype=float)

for i in range(m):
    inputs_random[i, :] = inputs[a[i], :]
    labels_random[i, :] = labels[a[i], :]

# Simpan hasil randomisasi
np.save(os.path.join(dir_out, "random_inputs_120_785.npy"), inputs_random)
np.save(os.path.join(dir_out, "random_labels_120_13.npy"),  labels_random)

print("✔ Randomisasi selesai. File disimpan di:", dir_out)
